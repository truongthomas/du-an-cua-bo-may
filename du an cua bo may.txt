# 將 HTML 寫入本地檔案
filename = "sfcs_dashboard.html"
with open(filename, "w", encoding="utf-8") as f:
   f.write(html_content)
PORT = 8080
class Handler(http.server.SimpleHTTPRequestHandler):
   def log_message(self, format, *args):
       # 覆寫 log_message 以隱藏終端機的 HTTP 請求紀錄，保持畫面乾淨
       pass
def start_server():
   with socketserver.TCPServer(("", PORT), Handler) as httpd:
       print(f"✅ 本地伺服器已啟動: http://localhost:{PORT}/{filename}")
       print("請按 Ctrl+C 來關閉伺服器。")
       httpd.serve_forever()
# 使用執行緒啟動伺服器，以便能自動開啟瀏覽器
server_thread = threading.Thread(target=start_server)
server_thread.daemon = True
server_thread.start()
# 等待一小段時間確保伺服器已啟動，然後開啟瀏覽器
time.sleep(1)
webbrowser.open(f"http://localhost:{PORT}/{filename}")
# 保持主程式運行
try:
   while True:
       time.sleep(1)
except KeyboardInterrupt:
   print("\\n🛑 伺服器已關閉。")
